# 1. 核心 NLP
# Stanza (StanfordNLP): 提供英文文本的分词、多词 token 展开 (MWT)、

stanza>=1.6.0

openai>=1.0.0

# 3. 数据处理

numpy>=1.24.0

scipy>=1.10.0
# scikit-learn: 精确率、召回率、F1 等分类指标。
scikit-learn>=1.3.0

# 5. 可视化
matplotlib>=3.7.0

# 6. 配置管理
# PyYAML: 将 model.yaml、prompts.yaml、constraints.yaml 解析为 Python 字典。
pyyaml>=6.0

# 7. 数据校验
# Pydantic v2: 运行时类型检查和校验。提前捕获格式错误的数据。
pydantic>=2.0.0

# 8. 进度显示
tqdm>=4.65.0
